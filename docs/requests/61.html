GET, HEAD              → specodex-api-read-<stage>
                         table.grantReadData(fn)
                         bucket.grantRead(fn)

# Write path
POST, PUT, PATCH,
DELETE                 → specodex-api-<stage>
                         table.grantReadWriteData(fn)
                         bucket.grantReadWrite(fn)
